### 方案二：智能推荐复习

#### 1. 基于考题概率推荐

```javascript
// 根据考题概率推荐复习顺序
function getRecommendedReviewOrder(userStats) {
  const recommendations = [
    {
      category: '密码法',
      priority: 5,
      reason: '核心必考，出题概率90%+',
      accuracy: userStats['密码法']?.accuracy || 0,
      questionCount: 95
    },
    {
      category: '网络安全法',
      priority: 5,
      reason: '核心必考，出题概率85%+',
      accuracy: userStats['网络安全法']?.accuracy || 0,
      questionCount: 80
    }
    // ...
  ];

  // 按优先级和准确率排序
  return recommendations.sort((a, b) => {
    if (a.priority !== b.priority) return b.priority - a.priority;
    return a.accuracy - b.accuracy; // 优先推荐准确率低的
  });
}
```

function generateSmartPath(userProgress) {

    type: 'medium_priority',

  });

}

## 🔧 数据库更新建议

### 1. 更新现有题目标注

```sql
-- 根据文档列表更新题目
UPDATE questions
SET
  original_document = CASE
    WHEN question_text LIKE '%密码法%' THEN '密码法'
    WHEN question_text LIKE '%网络安全法%' THEN '网络安全法'
    WHEN question_text LIKE '%SM2%' THEN 'SM2密码算法'
    -- ... 其他映射
  END,
  document_priority = CASE
    WHEN original_document IN ('密码法', '网络安全法', '数据安全法', '商用密码管理条例', 'SM2椭圆曲线公钥密码算法') THEN 5
    WHEN original_document IN ('个人信息保护法', '电子签名法', '传输层密码协议（TLCP）') THEN 5
    WHEN original_document IN ('SSLVPN技术规范', 'IPSecVPN技术规范') THEN 4
    -- ... 其他优先级
  END
WHERE original_document IS NULL;
```

### 2. 创建文档统计表

```sql
-- 创建文档统计表
CREATE TABLE document_stats (
  id SERIAL PRIMARY KEY,
  document_name VARCHAR(200) UNIQUE NOT NULL,
  document_category VARCHAR(100),
  document_priority INTEGER DEFAULT 1,
  total_questions INTEGER DEFAULT 0,
  practiced_questions INTEGER DEFAULT 0,
  correct_questions INTEGER DEFAULT 0,
  wrong_questions INTEGER DEFAULT 0,
  avg_accuracy DECIMAL(5,2),
  last_practiced_at TIMESTAMP,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 从现有题目统计生成
INSERT INTO document_stats (document_name, document_category, document_priority, total_questions)
SELECT
  original_document,
  '法律法规',
  document_priority,
  COUNT(*) as total_questions
FROM questions
WHERE original_document IS NOT NULL
GROUP BY original_document, document_priority;
```
